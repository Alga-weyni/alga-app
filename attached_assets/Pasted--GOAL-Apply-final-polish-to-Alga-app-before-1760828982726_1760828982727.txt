# 2Ô∏è‚É£  INVOICE PDF GENERATION (ERCA-COMPLIANT)
# File: server/src/utils/invoice.ts

import { jsPDF } from "jspdf";

export async function generateInvoice(booking, algaTIN, hostTIN) {
  const doc = new jsPDF();

  doc.setFontSize(16);
  doc.text("Alga Booking Invoice", 20, 20);
  doc.setFontSize(12);

  doc.text(`Booking ID: ${booking.id}`, 20, 35);
  doc.text(`Guest: ${booking.guestName}`, 20, 45);
  doc.text(`Property: ${booking.propertyName}`, 20, 55);
  doc.text(`Date: ${new Date(booking.createdAt).toLocaleDateString('en-ET')}`, 20, 65);

  doc.text(`Gross Amount: ${booking.totalAmount.toFixed(2)} ETB`, 20, 80);
  doc.text(`Alga Commission (12%): ${booking.algaCommission.toFixed(2)} ETB`, 20, 90);
  doc.text(`VAT (15%): ${booking.vat.toFixed(2)} ETB`, 20, 100);
  doc.text(`Withholding Tax (2%): ${booking.withholding.toFixed(2)} ETB`, 20, 110);
  doc.text(`Net Payout to Host: ${booking.hostPayout.toFixed(2)} ETB`, 20, 120);

  doc.text(`Alga TIN: ${algaTIN}`, 20, 140);
  doc.text(`Host TIN: ${hostTIN}`, 20, 150);
  doc.text("All taxes remitted per ERCA regulation.", 20, 165);

  return doc.output("datauristring");
}

export function formatETB(amount) {
  return new Intl.NumberFormat('en-ET', {
    style: 'currency',
    currency: 'ETB',
    minimumFractionDigits: 2,
    maximumFractionDigits: 2,
  }).format(amount);
}
