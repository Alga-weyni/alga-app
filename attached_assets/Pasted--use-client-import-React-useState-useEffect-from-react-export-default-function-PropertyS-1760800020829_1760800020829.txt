"use client";
import React, { useState, useEffect } from "react";

export default function PropertySearch() {
  // ğŸ§  State Variables
  const [city, setCity] = useState("");
  const [type, setType] = useState("");
  const [min, setMin] = useState("");
  const [max, setMax] = useState("");
  const [sort, setSort] = useState("recommended");
  const [search, setSearch] = useState("");
  const [results, setResults] = useState([]);
  const [loading, setLoading] = useState(false);

  // âš™ï¸ Fetch Properties
  async function fetchProperties() {
    setLoading(true);
    try {
      const res = await fetch(
        `/api/properties/search?city=${city}&type=${type}&minPrice=${min || 0}&maxPrice=${max || 9999999}&sort=${sort}&q=${search}`
      );
      const data = await res.json();
      setResults(data);
    } catch (err) {
      console.error("Search failed:", err);
    } finally {
      setLoading(false);
    }
  }

  // â³ Auto-refresh when filters change
  useEffect(() => {
    fetchProperties();
  }, [city, type, min, max, sort, search]);

  return (
    <div className="p-6 bg-[#fff9f2] min-h-screen">
      <h1 className="text-2xl font-semibold text-[#6b3f19] mb-6">
        Find Your Stay ğŸŒ¿
      </h1>

      {/* ğŸ” Filters */}
      <div className="flex flex-wrap gap-3 mb-5">
        <input
          type="text"
          placeholder="Search by keyword..."
          className="border border-[#d6a77a] rounded-md p-2 flex-1"
          value={search}
          onChange={(e) => setSearch(e.target.value)}
        />
        <input
          type="text"
          placeholder="City"
          className="border border-[#d6a77a] rounded-md p-2 w-40"
          value={city}
          onChange={(e) => setCity(e.target.value)}
        />
        <select
          className="border border-[#d6a77a] rounded-md p-2 w-40"
          value={type}
          onChange={(e) => setType(e.target.value)}
        >
          <option value="">All Types</option>
          <option value="guesthouse">Guesthouse</option>
          <option value="apartment">Apartment</option>
          <option value="villa">Villa</option>
          <option value="homestay">Homestay</option>
        </select>
        <input
          type="number"
          placeholder="Min Price (ETB)"
          className="border border-[#d6a77a] rounded-md p-2 w-32"
          value={min}
          onChange={(e) => setMin(e.target.value)}
        />
        <input
          type="number"
          placeholder="Max Price (ETB)"
          className="border border-[#d6a77a] rounded-md p-2 w-32"
          value={max}
          onChange={(e) => setMax(e.target.value)}
        />
        <select
          className="border border-[#d6a77a] rounded-md p-2 w-40"
          value={sort}
          onChange={(e) => setSort(e.target.value)}
        >
          <option value="recommended">Recommended</option>
          <option value="price_asc">Lowest Price</option>
          <option value="price_desc">Highest Price</option>
          <option value="rating_desc">Top Rated</option>
        </select>
      </div>

      {/* ğŸ  Results */}
      {loading ? (
        <p className="text-[#8c6239]">Loading properties...</p>
      ) : results.length === 0 ? (
        <p className="text-[#8c6239] italic">No properties found.</p>
      ) : (
        <div className="grid md:grid-cols-3 sm:grid-cols-2 grid-cols-1 gap-6">
          {results.map((p: any) => (
            <div
              key={p.id}
              className="border border-[#e2c4a3] rounded-lg shadow-sm p-4 bg-white hover:shadow-md transition-all"
            >
              <img
                src={p.image_url || "/placeholder.jpg"}
                alt={p.title}
                className="w-full h-40 object-cover rounded-md mb-3"
              />
              <h2 className="text-lg font-semibold text-[#6b3f19]">
                {p.title}
              </h2>
              <p className="text-sm text-gray-600">{p.city}</p>
              <p className="text-[#8c6239] font-medium mt-2">
                {p.price_per_night?.toLocaleString()} ETB / night
              </p>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
